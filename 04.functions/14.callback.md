# Callback function as argument
- Example:
```go
package main

import (
	"fmt"
)


func main() {
	fmt.Printf("%T\n", add)
	fmt.Printf("%T\n", subtract)
	fmt.Printf("%T\n", doMath)
	
	x := add(33,10)
	fmt.Println(x)

	y := subtract(33,15)
	fmt.Println(y)

	z := doMath(33, 10, add)
	fmt.Println(z)
	w := doMath(33, 10, subtract)
	fmt.Println(w)

}

func add(a, b int) int {
	return a + b
}

func subtract(a, b int) int {
	return a - b
}

func doMath(a int, b int, f func(int, int) int) int {
	return f(a,b)
}
```

Output:
```
func(int, int) int
func(int, int) int
func(int, int, func(int, int) int) int
43
18
43
23
```
